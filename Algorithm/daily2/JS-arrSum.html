function arrSum(arr) { // 모든 요소의 합을 리턴 // arr[0] + arr[1] + ... + arr[n-1] // arr.length는 n // 입력받은 배열은 함수의 호출 뒤에도 처음 상태를 유지해야 합니다(immutability). // 입력으로 들어오는 arr의 모든 요소는 정수 값을 갖는다고 가정합니다. // 빈 배열의 합은 0 입니다. // 1) [ ] => 0 // 2) [3] => 3 + [ ] // 3) [2, 3] => 2 + [3] // 4) [1, 2, 3] => 1 + [2, 3] if(arr.length === 0) return 0; // 빈배열 경우 // const head = arr[0]; // 0번째 head 할당 // const tail = arr.slice(1); // 0번재 제외 하고 나머지 tail 할당 // return head + arrSum(tail); // base case 될때 끼지 return arr.shift()+arrSum(arr) }
