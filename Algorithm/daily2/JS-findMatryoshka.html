function findMatryoshka(matryoshka, size) { // 객체와 수 입력 // 조건에 맞는지 ? // matryoshka.matryoshka는 null 또는 matryoshka 객체 // matryoshka.size는 중첩될수록 작아짐 if(matryoshka.size === size) return true; else if (matryoshka.matryoshka && matryoshka.size > size) { return findMatryoshka(matryoshka.matryoshka, size); } else { return false; } }
