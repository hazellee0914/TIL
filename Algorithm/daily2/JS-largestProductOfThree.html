const largestProductOfThree = function (arr) { // 정수를 요소로 갖는 배열을 입력 받기 // 3개의 요소로 곱해 나올 수 있는 최대값 리턴 // 42 (= 2 * 3 * 7) // 배열은 중첩되지 않은 1차원 배열 // 배열의 요소는 음수와 0을 포함하는 정수 // 배열의 길이 3이상 let list = arr.slice().sort((a, b) => a - b); const a1 = list[list.length-1]*list[list.length-2]*list[list.length-3]; const a2 = list[0]*list[1]*list[list.length-1]; return Math.max(a1,a2) };
