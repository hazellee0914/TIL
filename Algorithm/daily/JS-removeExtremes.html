function removeExtremes(arr) { // 가장 짧은 길이의 인덱스 값 // 그 인덱스 값 제외하고 하나씩 푸쉬 // splice()?? let min = arr[0]; let max = arr[0]; for (i in arr) { if(arr[i].length <= min.length) min = arr[i]; if(arr[i].length >= max.length) max = arr[i]; } arr.splice(arr.indexOf(min), 1); arr.splice(arr.indexOf(max), 1); return arr; }
