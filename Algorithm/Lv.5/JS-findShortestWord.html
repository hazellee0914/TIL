function findShortestWord(arr) { // 가장 짧은 길이 가진 문자열 요소 리턴 // 같은 길이 요소가 있다면 앞쪽 요소 리턴 // 문자열 외에 다른 요소가 있을 수 있음 // 빈 배열 경우 빈문자열로 리턴 => '' // 문자열이 없는 경우 빈문자열로 리턴 // 문자열로 구성된 배열 let str = arr.filter(el => typeof el === 'string'); if (str.length === 0) return ''; return str.reduce((acc, cur) => { if(acc.length <= cur.length) return acc; else return cur; }); }
