function studentReports(students) { // 남학생들이 정보는 리턴하는 배열에서 제외 // grades 속성값과 평균값 number 바꿈 let filtered = students.filter((el) => { return el.gender === 'female'; }); return filtered.map((el) => { let sum = el.grades.reduce((acc, cur) => { return acc + cur; }); let avg = sum / el.grades.length; el.grades = avg; return el; }); }
