function solution(n, arr1, arr2) { const answer = []; for(let i = 0; i < n; i++) { let result = ''; let b1 = '0'.repeat(n - arr1[i].toString(2).length) + arr1[i].toString(2); let b2 = '0'.repeat(n - arr2[i].toString(2).length) + arr2[i].toString(2); for(let j =0; j < n; j++) { b1[j] === '1' || b2[j] === '1' ? result += '#' : result += ' '; } answer.push(result); } return answer; }
