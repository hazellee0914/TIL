function getValueOfNthElement(arr, num) { // 배열과 수를 입력 // 빈 배열 no name -> 배열 arr 값의 길이가 하나도 없으면 no name // 배열 arr 값 num 보다 커야 함 // 수가 가리키는 마지막 객체 name 속성값 리턴 if (arr.length == 0) { return 'no name'; } else if ( num >= arr.length) { // 마지막 객체 name 값 return arr[arr.length -1]['name']; } else { return arr[num]['name']; } }
