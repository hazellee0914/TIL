function getEvenNumbers(arr) { // range 모듈과 비슷함 // 1. 짝수만을 요소로 갖는 배열을 만들어야하니 new 빈 배열을 만든다. // 2. 배열을 순회(반복문)하면서 짝수만 골라서 넣을거다 -> push 사용 // 2-1 짝수만 골라 넣는다 -> 2로 나눠 나머지가 0인 것 // 3. 배열을 리턴 let result = []; for (let i = 0; i < arr.length; i++) { if (arr[i] % 2 === 0) { result.push(arr[i]) } } return result; }
