function addOneSecond(hour, minute, second) { // 시 분 초 입력 // 1초 더한 결과 // 특정 메시지 리턴 if(second < 59) { return `1초 뒤에 ${hour}시 ${minute}분 ${second + 1}초 입니다`; } else { if(minute < 59){ return `1초 뒤에 ${hour}시 ${minute + 1}분 0초 입니다`; } else { if(hour < 23) { return `1초 뒤에 ${hour + 1}시 0분 0초 입니다`; } else { return `1초 뒤에 0시 0분 0초 입니다`; } } } }
