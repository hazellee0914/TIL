function getType(anything) { // typeof : 값의 타입 // object : 배열 타입 // Array.isArray : 배열과 객체를 구분시 // typeof를 사용하여 배열을 검사하면, typeof는 'object'를 리턴. // 배열은 'object'의 특수한 한 형태이기 때문 // typeof만으로는 객체가 배열인지 확인할 수 없다 // 객체가 배열인지 확인하기 위해서는 'isArray()' 함수를 사용 if (Array.isArray(anything) === true) { return 'array'; } if (anything === null) { retrun 'null'; } return typeof(anything); }
